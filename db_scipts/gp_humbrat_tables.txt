CREATE TABLE `db_gp_humbrat`.`tbl_news_panel` (
  `tbl_news_id` INT NOT NULL AUTO_INCREMENT,
  `tbl_news_title` VARCHAR(100) NOT NULL,
  `tbl_news_desciption` VARCHAR(10000) NOT NULL,
  `tbl_news_created_date` DATE NULL,
  `tbl_news_is_active` VARCHAR(10) BINARY NOT NULL,
  PRIMARY KEY (`tbl_news_id`));
  
ALTER TABLE `db_gp_humbrat`.`tbl_news_panel` 
ADD COLUMN `tbl_news_updated_date` DATE NULL AFTER `tbl_news_is_active`,
ADD COLUMN `tbl_news_is_deleted` VARCHAR(10) BINARY NOT NULL AFTER `tbl_news_updated_date`;

ALTER TABLE `db_gp_humbrat`.`tbl_news_panel` 
CHANGE COLUMN `tbl_news_is_active` `tbl_news_is_active` TINYINT NOT NULL ;


/* store procedure
09/07/2020
*/
USE `db_gp_humbrat`;
DROP procedure IF EXISTS `sp_newsAddUpdate`;

DELIMITER $$
USE `db_gp_humbrat`;
DROP procedure IF EXISTS `sp_newsAddUpdate`;

DELIMITER $$
USE `db_gp_humbrat`$$
CREATE DEFINER=`root`@`localhost` PROCEDURE `sp_newsAddUpdate`(
IN _tbl_news_id INT,
IN _tbl_news_title varchar(100),
IN _tbl_news_desciption varchar(10000),
IN _tbl_news_is_active tinyint
)
BEGIN
Declare createdUpdatedDate date;
if _tbl_news_is_active=1 then
 update tbl_news_panel
 set tbl_news_is_active=0
 where tbl_news_is_active=1;
 end if;
 IF _tbl_news_id=0 THEN
 set createdUpdatedDate=(SELECT CURDATE());
   INSERT INTO `db_gp_humbrat`.`tbl_news_panel`
(
`tbl_news_title`,
`tbl_news_desciption`,
`tbl_news_created_date`,
`tbl_news_is_active`,
`tbl_news_updated_date`,
`tbl_news_is_deleted`)
VALUES
(_tbl_news_title,_tbl_news_desciption,createdUpdatedDate,_tbl_news_is_active,createdUpdatedDate,0);
else
set createdUpdatedDate=(SELECT CURDATE());
UPDATE tbl_news_panel 
SET 
    tbl_news_title = _tbl_news_title,
    tbl_news_desciption = _tbl_news_desciption,
    tbl_news_is_active = _tbl_news_is_active,
    tbl_news_updated_date = createdUpdatedDate
WHERE
    tbl_news_id = _tbl_news_id;

end if;
END$$

DELIMITER ;

DELIMITER ;

DELIMITER $$
USE `db_gp_humbrat`$$
CREATE DEFINER=`root`@`localhost` PROCEDURE `sp_deleteNewsInfo`(
IN _tbl_news_id int
)
BEGIN
	update tbl_news_panel set tbl_news_is_deleted=1 where tbl_news_id=_tbl_news_id;
    update tbl_news_panel set tbl_news_is_active=0 where tbl_news_id=_tbl_news_id;
END$$
